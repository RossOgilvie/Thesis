\foreach \xx in {0.1, 0.4, 0.7, 1, 1.5, 2.5 , 10}
    \addplot[domain=0:180,samples=100,color=blue] ({ (\xx^2 - 1)/(\xx^2 + 2*\xx*sin(x) + 1) },{ (-2*\xx*cos(x))/(\xx^2 + 2*\xx*sin(x) + 1) });
